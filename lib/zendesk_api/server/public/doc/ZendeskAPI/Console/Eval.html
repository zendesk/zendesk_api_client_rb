def loop_eval(str)
  split = str.split(/\s*\/\s*/)
  split[1..-1] = split[1..-1].map do |s|
    s =~ /\d+/ ? s : "\"#{s}\""
  end
  split = split.join('/')
  split += '/' if str[-1] == '/'
  # puts ZendeskAPI::Console::ZD_DIRUP
  split.gsub!(/\.\./, "ZendeskAPI::Console::ZD_DIRUP")
  super(split)
end
