def votes(opts = {})
  return @votes if @votes && !opts[:reload]

  association = ZendeskAPI::Association.new(:class => TopicVote, :parent => self, :path => 'votes')
  @votes = ZendeskAPI::Collection.new(@client, TopicVote, opts.merge(:association => association))
end
